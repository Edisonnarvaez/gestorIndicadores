from rest_framework import serializers
from .models.company import Company
from .models.department import Department

# Serializador para el modelo Company
class CompanySerializer(serializers.ModelSerializer):
    class Meta:
        model = Company
        fields = ['companyId', 'name', 'nit', 'legalRepresentative', 'phone', 'address', 'contactEmail', 'foundationDate', 'status']

# Serializador para el modelo Department
class DepartmentSerializer(serializers.ModelSerializer):
    company = CompanySerializer(read_only=True)  # Si deseas mostrar información de la compañía en las respuestas

    class Meta:
        model = Department
        fields = ['departmentId', 'name', 'departmentCode', 'companyId', 'description', 'creationDate', 'status']
